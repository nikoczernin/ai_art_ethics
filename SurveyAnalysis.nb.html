ans <- read.delim("answers.csv", sep = ";")
names(ans) <- c("gender", "age", "program", "a1", "a2", "a3")
ans <-
  ans %>% 
  # rename the answer cols and transform them 
  mutate(
    revenue.deserved = str_extract(a1, "[:digit:]*") %>% as.numeric(),
    is.artist = a2 %>% as.factor(),
    would.boycott = a3 %>% as.factor()
    ) %>% 
  select(-a1, -a2, -a3)

ans %>% 
  drop_na() %>% 
  group_by(is.artist) %>% 
  summarise(
    `Average percentage of revenue granted to an AI-trained-on artist` = mean(revenue.deserved),
    `Total number of participants`=n()
  ) %>% 
  pivot_longer(c(`Total number of participants`,
                 `Average percentage of revenue granted to an AI-trained-on artist`
                 )) %>% 
  ggplot(aes(x=is.artist, y=value)) +
  geom_bar(stat="identity") +
    facet_wrap(~name, ncol = 1, scale="free") +
  xlab("Participants grouped by their relationship to creating art") +
  ylab(NULL)
